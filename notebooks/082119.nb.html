main code (setuprot2.R):

prof=list(0)
profr=list(0)
cyl_height=4
nstsh=4000
j=1
for(i in ind){
  
  snap0=read_nbody(i,Omega)
  
  snap=snap0$snapshot[[1]]
  
  #profiles los z-axis
  prof[[j]]=lag_analysis(snap,cyl_height,nstsh)
  
  r=sqrt(snap$x^2+snap$y^2+snap$z^2)
  
  #cylindrical los r<1.5rh
  sub=subset(snap,r<1.5*snap0$rh[1])
  
  angm=angmom(sub)
  Lx=sum(angm[,1]*angm[,4])
  Ly=sum(angm[,2]*angm[,4])
  Lz=sum(angm[,3]*angm[,4])
  
  I=momin(sub)
  L=c(Lx,Ly,Lz)
  
  om=solve(I,L)
  omx=om[1]
  omy=om[2]
  omz=om[3]
  
  phiom=atan2(omy,omx)*180/pi
  thetaom=acos(omz/sqrt(omx^2+omy^2+omz^2))*180/pi
  
  rot2=rotate(0,-phiom,snap)
  profr[[j]]=lag_analysis(rotate(-thetaom,0,rot2),cyl_height,nstsh) #cylindrical los r<1.5rh
  j=j+1
}
